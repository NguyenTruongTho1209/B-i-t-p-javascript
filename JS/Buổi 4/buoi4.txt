function            là 1 hàm được định nghĩa trước làm 1 chức năng cụ thể
                    function sẽ thường có đầu vào (input) và đầu ra (output)

khai báo function

function ten (input) {          'function' từ khóa khai báo
    // logic xử lý đầu vào      logic tự viết theo yêu cầu
    return output               return để trả về kết quả
}                               logic của functiong có thể tái sử dụng
